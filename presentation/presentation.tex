\section{Design} 
\subsection{The structure of the application}
\frame{
  \frametitle{Design}
  A CGI program based on socket and HTML5.
  \begin{itemize}
  \item
    \textbf{Server Side}\\
    Mojolicious, the Perl web framework
  \item
    \textbf{Client Side}\\
    HTML5 canvas and jQuery
  \item
    \textbf{Communication}\\
    JSON (JavaScript Object Notation)
  \item
    \textbf{User Interface}\\
    HTML with CSS and jQuery UI
  \end{itemize}
}

\frame{
  \frametitle{Design - Server Side}
  About 500 lines of Perl code.
  \begin{itemize}
  \item
    \textbf{Receiving Messages}\\
    Receive the messages sent by the clients
  \item
    \textbf{Parsing Messages}\\
    Perform different tasks according to the contents of messages
  \item
    \textbf{Database Manipulation}\\
    A database to store the line segments and chatting messages
  \item
    \textbf{Sending Messages Back}\\
    Send messages to the clients
  \end{itemize}
}

\frame{
  \frametitle{Design - Client Side}
  About 1,000 lines of Javascript, with the help of jQuery library
  \begin{itemize}
  \item
    \textbf{Initializing Connections}\\
    Establishing connections with the server
  \item
    \textbf{Detecting Event}\\
    Detecting and responding to mouse events
  \item
    \textbf{Sending Messages}\\
    Send the messages to the server
  \item
    \textbf{Receiving Data}\\
    Getting data from the server and performing correspondingly
  \item
    \textbf{Displaying Data}\\
    Drawing on the canvas and displaying chatting messages
  \end{itemize}
}

\frame{
  \frametitle{Design - Communication}
  Making use of JSON for data communication
  \begin{itemize}
  \item
    \textbf{Stringifying (Encode)}\\
    Store the data in an object into a string
  \item
    \textbf{Sending through WebSocket}\\
    Using the WebSocket to send between the server and clients
  \item
    \textbf{Parsing String}\\
    Parse the strings to get the data objects
  \end{itemize}
}

\frame{
  \frametitle{Design - User Interface}
  More than 300 lines of HTML, CSS and more Javascript to control UI
  \begin{itemize}
  \item
    \textbf{Simple and Elegant}\\
    GoogleDocs style appearance
  \item
    \textbf{jQuery UI}\\
    Making use of the jQuery UI library
  \end{itemize}
}

\subsection{Multiple Canvas}
\frame{
  \frametitle{Highlights - Multiple Canvas}
  Undo and Redo, requires three types of canvas for different tasks\pause
  \begin{itemize}
  \item
    \textbf{Base Canvas}\\
    For drawing the "static" segments\pause
  \item
    \textbf{User's Layers}\\
    For drawing the tentative segments\pause
  \item
    \textbf{Detector}\\
    Detecting the mouse events\pause
  \end{itemize}
  Arrangement of layers need to be changed frequently\pause
  \begin{itemize}
  \item
    \textbf{User Logging In and Out}\\
    Inserting and deleting canvases\pause
  \item
    \textbf{Drawing}\\
    Rearranging the order of canvases
  \end{itemize}
}
